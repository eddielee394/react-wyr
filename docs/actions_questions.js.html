import { API } from "../utils/api";

export const RECEIVE_QUESTIONS = "RECEIVE_QUESTIONS";
export const VOTE_QUESTION_ANSWER = "VOTE_QUESTION_ANSWER";
export const SET_VISIBILITY_FILTER = "SET_VISIBILITY_FILTER";

/**
 * Constants
 */
export const VisibilityFilters = {
  SHOW_ALL: "SHOW_ALL",
  SHOW_ANSWERED: "SHOW_ANSWERED",
  SHOW_UNANSWERED: "SHOW_UNANSWERED"
};

/**
 * Receive Questions action type
 * @param questions
 * @return {{type: string, questions: *}}
 */
export const receiveQuestions = questions => ({
  type: RECEIVE_QUESTIONS,
  questions
});

export const voteQuestionAnswer = ({ authUser, questionId, answer }) => ({
  type: VOTE_QUESTION_ANSWER,
  authUser,
  questionId,
  answer
});

/**
 * Set visiblity filter
 * @param filter
 * @return {{type: string, filter: *}}
 */
export const setVisiblityFilter = filter => ({
  type: SET_VISIBILITY_FILTER,
  filter
});

/**
 * Handles saving vote for a question answer
 * @param params
 * @return {function(*): Promise<any | never>}
 */
export const handleVoteQuestionAnswer = params => dispatch => {
  dispatch(voteQuestionAnswer(params));
  return API.saveQuestionAnswer(params).catch(event => {
    console.warn("Error in handleToggleTweet: ", event);
    dispatch(voteQuestionAnswer(params));
    //todo add toast notification
  });
};
